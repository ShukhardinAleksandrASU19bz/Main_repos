#include <SoftwareSerial.h>
#include <Wire.h>

// C++ code
//
#define RED 13  // присваиваем имя RED для пина 11
#define GRN 11 // присваиваем имя GRN для пина 12
#define BLU 12  // присваиваем имя BLU для пина 13
#define PIR 10 
int SensorTemp = 0;

void setup()
{
  pinMode(LED_BUILTIN, OUTPUT);
  pinMode(RED, OUTPUT);  // используем Pin11 для вывода
  pinMode(GRN, OUTPUT); // используем Pin12 для вывода
  pinMode(BLU, OUTPUT);  // используем Pin13 для вывода
  pinMode(PIR, INPUT);
}

void loop()
{
  Serial.begin(9600);  
  int readPIR = digitalRead(PIR);
  int readtemp = analogRead(SensorTemp);
  float voltage = readtemp * 5;
  voltage /=  1024;
  float temperatureC = (voltage - 0.5) * 100;
  
  //обнаружение движения
  if (readPIR)
  {
    Serial.print(readPIR);
    Serial.println("Motion detected");
    if (temperatureC < 10) 
    { 
      digitalWrite(RED, LOW);
      digitalWrite(GRN, LOW);
      digitalWrite(BLU, HIGH);
   	}
    if (temperatureC > 10 && temperatureC < 60) 
    { 
      digitalWrite(RED, LOW);
      digitalWrite(GRN, HIGH);
      digitalWrite(BLU, LOW);
    }
  	if (temperatureC > 60) 
    { 
      digitalWrite(RED, HIGH);
      digitalWrite(GRN, LOW);
      digitalWrite(BLU, LOW);
    }
     delay(1000);
  }
  // нет движения
  else
  {
    Serial.print(readPIR);
    Serial.println("No motion");
    digitalWrite(RED, LOW);
    digitalWrite(GRN, LOW);
    digitalWrite(BLU, LOW);
  }
  
  Serial.print(temperatureC); 
  Serial.println(" C");
  delay(1000); // Wait for 1000 millisecond(s)
}
