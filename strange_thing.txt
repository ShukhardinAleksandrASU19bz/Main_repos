#include <SoftwareSerial.h>
#include <Wire.h>

// C++ code
//
#define RED 11  // присваиваем имя RED для пина 11
#define GRN 12 // присваиваем имя GRN для пина 12
#define BLU 13  // присваиваем имя BLU для пина 13
int SensorTemp = 0;


void setup()
{
  pinMode(LED_BUILTIN, OUTPUT);
  pinMode(RED, OUTPUT);  // используем Pin11 для вывода
  pinMode(GRN, OUTPUT); // используем Pin12 для вывода
  pinMode(BLU, OUTPUT);  // используем Pin13 для вывода
}

void loop()
{
  Serial.begin(9600);  
  digitalWrite(LED_BUILTIN, HIGH);
  int readtemp = analogRead(SensorTemp);
  float voltage = readtemp * 5;
  voltage /=  1024;
  float temperatureC = (voltage - 0.5) * 100;
  if (temperatureC < 10) { digitalWrite(RED, HIGH);
    digitalWrite(GRN, LOW);
    digitalWrite(RED, LOW);
   }
  if (temperatureC > 10 && temperatureC < 60) { digitalWrite(RED, LOW);
    digitalWrite(GRN, HIGH);
    digitalWrite(RED, LOW);
    }
  if (temperatureC > 60) { digitalWrite(RED, LOW);
    digitalWrite(GRN, LOW);
    digitalWrite(RED, HIGH);
    }
 // Serial.print(temperatureC); 
 // Serial.println(" C");

}
